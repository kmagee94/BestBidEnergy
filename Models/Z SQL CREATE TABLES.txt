CREATE TABLE BUYER (
USERNAME VARCHAR(255) PRIMARY KEY,
EMAIL VARCHAR(255),
PHONE VARCHAR(255),
COMPANY VARCHAR(255),
ACCOUNT_NUM VARCHAR(255),
ACCOUNT_TYPE VARCHAR(255),
CREDIT_RATING INT,
CONTACT_NAME VARCHAR(255)
)

CREATE TABLE SUPPLIER (
USERNAME VARCHAR(255) PRIMARY KEY,
EMAIL VARCHAR(255),
PHONE VARCHAR(255),
ACCOUNT_NUM VARCHAR(255),
COMPANY VARCHAR(255),
CONTACT_NAME VARCHAR(255)
)

CREATE TABLE CONTRACTS(
CONTRACT_ID INT PRIMARY KEY,
BUYER_USERNAME VARCHAR(255) FOREIGN KEY REFERENCES BUYER(USERNAME),
SUPPLIER VARCHAR(255) FOREIGN KEY REFERENCES SUPPLIER(USERNAME),
PRICE FLOAT,
STREET VARCHAR(255),
CITY VARCHAR(50),
[STATE] VARCHAR(40),
ZIP INT,
UTILITY_AREA VARCHAR(255),
STARTDATE DATE,
ENDDATE DATE,
AVG_kWh INT,
UTILITY_ACCOUNT VARCHAR(255)
)

CREATE TABLE AUCTION(
AUCTION_ID INT PRIMARY KEY,
AUCTION_TIME TIME,
CONTRACT_ID INT FOREIGN KEY REFERENCES CONTRACTS(CONTRACT_ID)
)
